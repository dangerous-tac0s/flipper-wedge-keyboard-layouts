Name: Dvorak

# Dvorak keyboard layout
# Alternative ergonomic layout
# Numbers are in the same positions as QWERTY
# Letters are completely rearranged
#
# Dvorak letter positions:
# Top row:    ' , . p y f g c r l
# Home row:   a o e u i d h t n s
# Bottom row: ; q j k x b m w v z
#
# HID keycodes for hex letters on Dvorak:
# To produce a character on Dvorak, send the keycode for the QWERTY key
# that is in that position on a Dvorak layout.
#
# A = 'a' is at QWERTY 'a' position = keycode 4
# B = 'b' is at QWERTY 'n' position = keycode 17
# C = 'c' is at QWERTY 'i' position = keycode 12
# D = 'd' is at QWERTY 'h' position = keycode 11
# E = 'e' is at QWERTY 'd' position = keycode 7
# F = 'f' is at QWERTY 'y' position = keycode 28

# Digits (same positions as QWERTY)
